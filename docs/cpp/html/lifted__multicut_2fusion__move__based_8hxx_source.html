204  template<class OBJECTIVE>
205  void FusionMoveBased<OBJECTIVE>::
206  optimizeSingleThread(
207  VisitorProxyType & visitorProxy
208  ){
209  NodeLabelsType proposal(graph_);
210  auto iterWithoutImprovement = 0;
211 
212  for(auto iteration=0; iteration<settings_.numberOfIterations; ++iteration){
213 
214  // generate a proposal
215  proposalGenerator_->generateProposal(*currentBest_, proposal, 0);
216 
217  // eval energy of proposal
218  //std::cout<<"E "<<objective_.evalNodeLabels(proposal)<<"\n";
219 
220  // accept the first proposal as current best
221  // if starting point was trivial (one connected comp.)
222  if(currentBestEnergy_ >=-0.000000001 && iteration==0){
223  graph_.forEachNode([&](const uint64_t node){
224  (*currentBest_)[node] = proposal[node];
225  });
226  currentBestEnergy_ = objective_.evalNodeLabels(*currentBest_);
227  }
228  else{
229  NodeLabelsType res(graph_);
230  auto & fm = *(fusionMoves_[0]);
231  fm.fuse( {&proposal, currentBest_}, &res);
232  auto eFuse = objective_.evalNodeLabels(res);
233  if(eFuse<currentBestEnergy_){
234  currentBestEnergy_ = eFuse;
235  graph_.forEachNode([&](const uint64_t node){
236  (*currentBest_)[node] = res[node];
237  });
238  iterWithoutImprovement = 0;
239  }
240  else{
241  ++iterWithoutImprovement;
242  if(iterWithoutImprovement >= settings_.stopIfNoImprovement){
243  break;
244  }
245  }
246  }
247 
248  visitorProxy.visit(this);
249  }
