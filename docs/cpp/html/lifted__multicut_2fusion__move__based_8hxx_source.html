203  template<class OBJECTIVE>
204  void FusionMoveBased<OBJECTIVE>::
205  optimizeSingleThread(
206  VisitorProxyType & visitorProxy
207  ){
208  NodeLabelsType proposal(graph_);
209  auto iterWithoutImprovement = 0;
210 
211  for(auto iteration=0; iteration<settings_.numberOfIterations; ++iteration){
212 
213  // generate a proposal
214  proposalGenerator_->generateProposal(*currentBest_, proposal, 0);
215 
216  // eval energy of proposal
217  //std::cout<<"E "<<objective_.evalNodeLabels(proposal)<<"\n";
218 
219  // accept the first proposal as current best
220  // if starting point was trivial (one connected comp.)
221  if(currentBestEnergy_ >=-0.000000001 && iteration==0){
222  graph_.forEachNode([&](const uint64_t node){
223  (*currentBest_)[node] = proposal[node];
224  });
225  currentBestEnergy_ = objective_.evalNodeLabels(*currentBest_);
226  }
227  else{
228  NodeLabelsType res(graph_);
229  auto & fm = *(fusionMoves_[0]);
230  fm.fuse( {&proposal, currentBest_}, &res);
231  auto eFuse = objective_.evalNodeLabels(res);
232  if(eFuse<currentBestEnergy_){
233  currentBestEnergy_ = eFuse;
234  graph_.forEachNode([&](const uint64_t node){
235  (*currentBest_)[node] = res[node];
236  });
237  iterWithoutImprovement = 0;
238  }
239  else{
240  ++iterWithoutImprovement;
241  if(iterWithoutImprovement >= settings_.stopIfNoImprovement){
242  break;
243  }
244  }
245  }
246 
247  visitorProxy.visit(this);
248  }
