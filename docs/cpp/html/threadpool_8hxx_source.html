209 inline void ThreadPool::init(const ParallelOptions & options)
210 {
211  const size_t actualNThreads = options.getNumThreads();
212  for(size_t ti = 0; ti<actualNThreads; ++ti)
213  {
214  workers.emplace_back(
215  [ti,this]
216  {
217  for(;;)
218  {
219  std::function<void(int)> task;
220  {
221  std::unique_lock<std::mutex> lock(this->queue_mutex);
222 
223  // will wait if : stop == false AND queue is empty
224  // if stop == true AND queue is empty thread function will return later
225  //
226  // so the idea of this wait, is : If where are not in the destructor
227  // (which sets stop to true, we wait here for new jobs)
228  this->worker_condition.wait(lock, [this]{ return this->stop || !this->tasks.empty(); });
229  if(!this->tasks.empty())
230  {
231  ++busy;
232  task = std::move(this->tasks.front());
233  this->tasks.pop();
234  lock.unlock();
235  task(ti);
236  ++processed;
237  --busy;
238  finish_condition.notify_one();
239  }
240  else if(stop)
241  {
242  return;
243  }
244  }
245  }
246  }
247  );
248  }
