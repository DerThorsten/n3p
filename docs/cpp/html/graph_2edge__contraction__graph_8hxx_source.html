621  const auto deadNode = aliveNode == u ? v : u;

623  NIFTY_ASSERT_OP(nodeUfd_.find(deadNode),!=,deadNode);

630  auto & adjDead = nodes_[deadNode];

632  // remove them from each other
633  //
634  //

636  NIFTY_ASSERT(adjDead.find(NodeAdjacency(aliveNode)) != adjDead.end());

639  adjDead.erase(NodeAdjacency(aliveNode));
