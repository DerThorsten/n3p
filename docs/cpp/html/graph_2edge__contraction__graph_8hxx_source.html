608  ){
609 
610  //
611  callback_.contractEdge(edgeToContract);
612  --currentEdgeNum_;
613 
614  // get the u and v we need to merge into a single node
615  const auto uv = this->uv(edgeToContract);
616  //const auto uv = edges_[edgeToContract];
617  const auto u = uv.first;
618  const auto v = uv.second;
619  NIFTY_TEST_OP(u,!=,v);
620 
621  // merge them into a single node
622  NIFTY_ASSERT_OP(nodeUfd_.find(u),==,u);
623  NIFTY_ASSERT_OP(nodeUfd_.find(v),==,v);
624  nodeUfd_.merge(u, v);
625  --currentNodeNum_;
626 
627  // check which of u and v is the new representative node
628  // also known as 'aliveNode' and which is the deadNode
629  const auto aliveNode = nodeUfd_.find(u);
630  NIFTY_ASSERT(aliveNode==u || aliveNode==v);
631  const auto deadNode = aliveNode == u ? v : u;
632  NIFTY_ASSERT_OP(nodeUfd_.find(aliveNode),==,aliveNode);
633  NIFTY_ASSERT_OP(nodeUfd_.find(deadNode),!=,deadNode);
634 
635  callback_.mergeNodes(aliveNode, deadNode);
636 
637 
638  // get the adjacency sets of both nodes
639  auto & adjAlive = nodes_[aliveNode];
640  auto & adjDead = nodes_[deadNode];
641 
642  // remove them from each other
643  //
644  //
645  NIFTY_ASSERT(adjAlive.find(NodeAdjacency(deadNode)) != adjAlive.end());
646  NIFTY_ASSERT(adjDead.find(NodeAdjacency(aliveNode)) != adjDead.end());
647 
648  adjAlive.erase(NodeAdjacency(deadNode));
649  adjDead.erase(NodeAdjacency(aliveNode));
650 
651 
652  // we will "shift/move" the adj. nodes
653  // from 'adjDead' into 'adjAlive':
654  for(auto adj : adjDead){
655 
656  const auto adjToDeadNode = adj.node();
657  const auto adjToDeadNodeEdge = adj.edge();
658 
659

662  const auto findResIter = adjAlive.find(NodeAdjacency(adjToDeadNode));
