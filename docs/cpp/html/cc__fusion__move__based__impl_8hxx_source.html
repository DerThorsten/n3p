199 
200  for(auto iteration=0; iteration<settings_.numberOfIterations; ++iteration){
201 
202  // generate a proposal
203  proposalGenerator_->generateProposal(*currentBest_, proposal, 0);
204 
205  if(currentBestEnergy_ >=-0.000000001 && iteration==0){
206  graph_.forEachNode([&](const uint64_t node){
207  (*currentBest_)[node] = proposal[node];
208  });
209  currentBestEnergy_ = objective_.evalNodeLabels(*currentBest_);
210  }
211  else{
212  NodeLabels res(graph_);

214  fm.fuse( {&proposal, currentBest_}, &res);
215  auto eFuse = objective_.evalNodeLabels(res);
216  if(eFuse<currentBestEnergy_){
217  currentBestEnergy_ = eFuse;
218  graph_.forEachNode([&](const uint64_t node){
219  (*currentBest_)[node] = res[node];
220  });
221  iterWithoutImprovement = 0;
222  }
