402

404 
405  // merge the node adjacency sets for each node
406  nifty::parallel::parallel_foreach(threadpool, numberOfLabels, [&](int tid, int label){

408  for(size_t i=1; i<perThreadDataVec.size(); ++i){
409  const auto & setI = perThreadDataVec[i].adjacency[label];
410  set0.insert(setI.begin(), setI.end());
411  }
412  for(auto otherNode : set0)
413  this->nodes_[label].insert(NodeAdjacency(otherNode));
414  });

422  if(u < v){
423  edges.push_back(EdgeStorage(u, v));
424  vAdj.changeEdgeIndex(edgeIndex);
425  auto fres = this->nodes_[v].find(NodeAdjacency(u));
426  fres->changeEdgeIndex(edgeIndex);

428  }

449 UndirectedGraph<EDGE_INTERANL_TYPE, NODE_INTERNAL_TYPE>
450 UndirectedGraph<EDGE_INTERANL_TYPE, NODE_INTERNAL_TYPE>::
451 extractSubgraphFromNodes(
452  const marray::View<NODE_INTERNAL_TYPE> & nodeList,
453  std::vector<EDGE_INTERANL_TYPE> & innerEdgesOut,
454  std::vector<EDGE_INTERANL_TYPE> & outerEdgesOut) const {
455 
456  std::map<NodeInteralType,NodeInteralType> globalToLocalNodes;
457  for(size_t i = 0; i < nodeList.size(); i++)
458  globalToLocalNodes.insert( std::make_pair(nodeList(i), NodeInteralType(i)) );
459 
460  for(auto nodeIt = nodeList.begin(); nodeIt != nodeList.end(); ++nodeIt) {
461  auto u = *nodeIt;
462  for(auto adjacencyIt = this->adjacencyBegin(u); adjacencyIt != this->adjacencyEnd(u); ++adjacencyIt) {
463  auto v = adjacencyIt->node();
464  auto e = this->findEdge(u, v);
465  if( std::find(nodeList.begin(), nodeList.end(), v) != nodeList.end() )
466  innerEdgesOut.push_back(e);
467  else
468  outerEdgesOut.push_back(e);
469  }
470  }
471 
472  // make the edges unique
473  std::sort(innerEdgesOut.begin(),innerEdgesOut.end());
474  auto last = std::unique(innerEdgesOut.begin(), innerEdgesOut.end());
475  innerEdgesOut.erase( last, innerEdgesOut.end() );
476 
477  std::sort(outerEdgesOut.begin(),outerEdgesOut.end());
478  last = std::unique(outerEdgesOut.begin(), outerEdgesOut.end());
479  outerEdgesOut.erase( last, outerEdgesOut.end() );
480 
481  // get number of nodes
482  uint64_t numberOfNodes = nodeList.size();
483  UndirectedGraph<EdgeInternalType,NodeInteralType> subgraphOut(numberOfNodes);
484 
485  // get the local uv - ids
486  for(size_t i = 0; i < innerEdgesOut.size(); ++i) {
487  auto uv = this->uv(innerEdgesOut[i]);
488  subgraphOut.insertEdge(globalToLocalNodes[uv.first], globalToLocalNodes[uv.second]);
489  }
