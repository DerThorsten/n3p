402 ){

404 
405 
406 
407  // merge the node adjacency sets for each node
408  nifty::parallel::parallel_foreach(threadpool, numberOfLabels, [&](int tid, int label){
409  auto & set0 = perThreadDataVec[0].adjacency[label];
410  for(std::size_t i=1; i<perThreadDataVec.size(); ++i){
411  const auto & setI = perThreadDataVec[i].adjacency[label];
412  set0.insert(setI.begin(), setI.end());
413  }
414  for(auto otherNode : set0)
415  this->nodes_[label].insert(NodeAdjacency(otherNode));
416  });
417 
418  // insert the edge index for each edge
419  uint64_t edgeIndex = 0;
420  auto & edges = this->edges_;
421  for(uint64_t u = 0; u< numberOfLabels; ++u){
422  for(auto & vAdj : this->nodes_[u]){
423  const auto v = vAdj.node();
424  if(u < v){
425  edges.push_back(EdgeStorage(u, v));
426  vAdj.changeEdgeIndex(edgeIndex);
427  auto fres = this->nodes_[v].find(NodeAdjacency(u));
428  fres->changeEdgeIndex(edgeIndex);
429  ++edgeIndex;
430  }

453 extractSubgraphFromNodes(
454  const marray::View<NODE_INTERNAL_TYPE> & nodeList,
455  std::vector<EDGE_INTERANL_TYPE> & innerEdgesOut,
456  std::vector<EDGE_INTERANL_TYPE> & outerEdgesOut) const {

459  for(std::size_t i = 0; i < nodeList.size(); i++)
460  globalToLocalNodes.insert( std::make_pair(nodeList(i), NodeInteralType(i)) );
461 
462  for(auto nodeIt = nodeList.begin(); nodeIt != nodeList.end(); ++nodeIt) {
463  auto u = *nodeIt;

465  auto v = adjacencyIt->node();
466  auto e = this->findEdge(u, v);
467  if( std::find(nodeList.begin(), nodeList.end(), v) != nodeList.end() )
468  innerEdgesOut.push_back(e);

471  }
472  }

476  auto last = std::unique(innerEdgesOut.begin(), innerEdgesOut.end());
477  innerEdgesOut.erase( last, innerEdgesOut.end() );
478 
479  std::sort(outerEdgesOut.begin(),outerEdgesOut.end());

481  outerEdgesOut.erase( last, outerEdgesOut.end() );
482

486
