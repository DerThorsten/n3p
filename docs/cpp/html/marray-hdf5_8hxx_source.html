230 template<class T, class ShapeIterator>
231 void create(
232  const hid_t& groupHandle,
233  const std::string& datasetName,
234  ShapeIterator begin,
235  ShapeIterator end,
236  CoordinateOrder coordinateOrder
237 ) {
238  marray_detail::Assert(MARRAY_NO_ARG_TEST || groupHandle >= 0);
239  HandleCheck<MARRAY_NO_DEBUG> handleCheck;
240 
241  // build dataspace
242  hid_t datatype = H5Tcopy(hdf5Type<T>());
243  std::size_t dimension = std::distance(begin, end);
244  std::vector<hsize_t> shape(dimension);
245  if(coordinateOrder == FirstMajorOrder) {
246  // copy shape as is
247  for(std::size_t j=0; j<dimension; ++j) {
248  shape[j] = hsize_t(*begin);
249  ++begin;
250  }
251  }
252  else {
253  // reverse shape
254  for(std::size_t j=0; j<dimension; ++j) {
255  shape[dimension-j-1] = hsize_t(*begin);
256  ++begin;
257  }
258  }
259  hid_t dataspace = H5Screate_simple(dimension, &shape[0], NULL);
260  if(dataspace < 0) {
261  H5Tclose(datatype);
262  throw std::runtime_error("Marray cannot create dataspace.");
263  }
264 
265 
266  // create new dataset
267  hid_t dataset = H5Dcreate(groupHandle, datasetName.c_str(), datatype,
268  dataspace, H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
269  if(dataset < 0) {
270  H5Sclose(dataspace);
271  H5Tclose(datatype);
272  throw std::runtime_error("Marray cannot create dataset.");
273  }
274 
275  // write attribute to indicate whether shape is reversed
276  if(coordinateOrder == LastMajorOrder) {
277  hsize_t attributeShape[1] = {1};
278  hid_t attributeDataspace = H5Screate_simple(1, attributeShape, NULL);
279  if(attributeDataspace < 0) {
280  H5Dclose(dataset);
281  H5Sclose(dataspace);
282  H5Tclose(datatype);
283  throw std::runtime_error("Marray cannot create dataspace.");
284  }
285  hid_t attribute = H5Acreate(dataset, reverseShapeAttributeName,
286  H5T_STD_U8LE, attributeDataspace, H5P_DEFAULT, H5P_DEFAULT);
287  if(attribute < 0) {
288  H5Sclose(attributeDataspace);
289  H5Dclose(dataset);
290  H5Sclose(dataspace);
291  H5Tclose(datatype);
292  throw std::runtime_error("Marray cannot create attribute.");
293  }
294  unsigned int data = 1;
295  herr_t err = H5Awrite(attribute, H5T_STD_U8LE, &data);
296  H5Aclose(attribute);
297  H5Sclose(attributeDataspace);
298  if(err < 0) {
299  H5Dclose(dataset);
300  H5Sclose(dataspace);
301  H5Tclose(datatype);
302  throw std::runtime_error("Marray cannot create write to attribute.");
303  }
304  }
305 
306  // clean up
307  H5Dclose(dataset);
308  H5Sclose(dataspace);
309  H5Tclose(datatype);
310  handleCheck.check();
311 }

321 template<class T>
322 void save(
323  const hid_t& groupHandle,
324  const std::string& datasetName,
325  const Marray<T>& in
326 ) {
327  marray_detail::Assert(MARRAY_NO_ARG_TEST || groupHandle >= 0);
328  HandleCheck<MARRAY_NO_DEBUG> handleCheck;
329 
330  // build dataspace
331  hid_t datatype = H5Tcopy(hdf5Type<T>());
332  std::vector<hsize_t> shape(in.dimension());
333  if(in.coordinateOrder() == FirstMajorOrder) {
334  // copy shape as is
335  for(std::size_t j=0; j<in.dimension(); ++j) {
336  shape[j] = hsize_t(in.shape(j));
337  }
338  }
339  else {
340  // reverse shape
341  for(std::size_t j=0; j<in.dimension(); ++j) {
342  shape[std::size_t(in.dimension()-j-1)] = hsize_t(in.shape(j));
343  }
344  }
345  hid_t dataspace = H5Screate_simple(in.dimension(), &shape[0], NULL);
346  if(dataspace < 0) {
347  H5Tclose(datatype);
348  throw std::runtime_error("Marray cannot create dataspace.");
349  }
350 
351  // create new dataset
352  hid_t dataset = H5Dcreate(groupHandle, datasetName.c_str(), datatype,
353  dataspace, H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
354  if(dataset < 0) {
355  H5Sclose(dataspace);
356  H5Tclose(datatype);
357  throw std::runtime_error("Marray cannot create dataset.");
358  }
359 
360  // write attribute to indicate whether shape is reversed
361  if(in.coordinateOrder() == LastMajorOrder) {
362  hsize_t attributeShape[1] = {1};
363  hid_t attributeDataspace = H5Screate_simple(1, attributeShape, NULL);
364  if(attributeDataspace < 0) {
365  H5Dclose(dataset);
366  H5Sclose(dataspace);
367  H5Tclose(datatype);
368  throw std::runtime_error("Marray cannot create dataspace.");
369  }
370  hid_t attribute = H5Acreate(dataset, reverseShapeAttributeName,
371  H5T_STD_U8LE, attributeDataspace, H5P_DEFAULT, H5P_DEFAULT);
372  if(attribute < 0) {
373  H5Sclose(attributeDataspace);
374  H5Dclose(dataset);
375  H5Sclose(dataspace);
376  H5Tclose(datatype);
377  throw std::runtime_error("Marray cannot create attribute.");
378  }
379  unsigned int data = 1;
380  herr_t err = H5Awrite(attribute, H5T_STD_U8LE, &data);
381  H5Aclose(attribute);
382  H5Sclose(attributeDataspace);
383  if(err < 0) {
384  H5Dclose(dataset);
385  H5Sclose(dataspace);
386  H5Tclose(datatype);
387  throw std::runtime_error("Marray cannot create write to attribute.");
388  }
389  }
390 
391  // write
392  herr_t status = H5Dwrite(dataset, datatype, H5S_ALL, H5S_ALL,
393  H5P_DEFAULT, &in(0));
394  H5Dclose(dataset);
395  H5Sclose(dataspace);
396  H5Tclose(datatype);
397  if(status < 0) {
398  throw std::runtime_error("Marray cannot write to dataset.");
399  }
400 
401  handleCheck.check();
402 }

523 template<class T>
524 void load(
525  const hid_t& groupHandle,
526  const std::string& datasetName,
527  Marray<T>& out
528 ) {
529  marray_detail::Assert(MARRAY_NO_ARG_TEST || groupHandle >= 0);
530  HandleCheck<MARRAY_NO_DEBUG> handleCheck;
531 
532  hid_t dataset = H5Dopen(groupHandle, datasetName.c_str(), H5P_DEFAULT);
533  if(dataset < 0) {
534  throw std::runtime_error("Marray cannot open dataset.");
535  }
536  hid_t filespace = H5Dget_space(dataset);
537  hid_t type = H5Dget_type(dataset);
538  hid_t nativeType = H5Tget_native_type(type, H5T_DIR_DESCEND);
539  if(!H5Tequal(nativeType, hdf5Type<T>())) {
540  H5Dclose(dataset);
541  H5Tclose(nativeType);
542  H5Tclose(type);
543  H5Sclose(filespace);
544  throw std::runtime_error("Data types not equal error.");
545  }
546  int dimension = H5Sget_simple_extent_ndims(filespace);
547  std::vector<hsize_t> shape(dimension);
548  herr_t status = H5Sget_simple_extent_dims(filespace, &shape[0], NULL);
549  if(status < 0) {
550  H5Dclose(dataset);
551  H5Tclose(nativeType);
552  H5Tclose(type);
553  H5Sclose(filespace);
554  throw std::runtime_error("H5Sget_simple_extent_dims error.");
555  }
556  hid_t memspace = H5Screate_simple(dimension, &shape[0], NULL);
557 
558  // resize marray
559  std::vector<std::size_t> newShape((std::size_t)(dimension));
560  for(std::size_t j=0; j<newShape.size(); ++j) {
561  newShape[j] = (std::size_t)(shape[j]);
562  }
563  if(H5Aexists(dataset, reverseShapeAttributeName) > 0) {
564  // reverse shape
565  out = Marray<T>(SkipInitialization, newShape.rbegin(),
566  newShape.rend(), LastMajorOrder);
567  }
568  else {
569  // don't reverse shape
570  out = Marray<T>(SkipInitialization, newShape.begin(),
571  newShape.end(), FirstMajorOrder);
572  }
573 
574  // read
575  status = H5Dread(dataset, nativeType, memspace, filespace,
576  H5P_DEFAULT, &out(0));
577  H5Dclose(dataset);
578  H5Tclose(nativeType);
579  H5Tclose(type);
580  H5Sclose(memspace);
581  H5Sclose(filespace);
582  if(status < 0) {
583  throw std::runtime_error("Marray cannot read from dataset.");
584  }
585 
586  handleCheck.check();
587 }

597 template<class T>
598 void loadShape(
599  const hid_t& groupHandle,
600  const std::string& datasetName,
601  std::vector<T>& out
602 ) {
603  marray_detail::Assert(MARRAY_NO_ARG_TEST || groupHandle >= 0);
604  HandleCheck<MARRAY_NO_DEBUG> handleCheck;
605 
606  // load shape from HDF5 file
607  hid_t dataset = H5Dopen(groupHandle, datasetName.c_str(), H5P_DEFAULT);
608  if(dataset < 0) {
609  throw std::runtime_error("Marray cannot open dataset.");
610  }
611  hid_t filespace = H5Dget_space(dataset);
612  hsize_t dimension = H5Sget_simple_extent_ndims(filespace);
613  hsize_t* shape = new hsize_t[(std::size_t)(dimension)];
614  herr_t status = H5Sget_simple_extent_dims(filespace, shape, NULL);
615  if(status < 0) {
616  H5Dclose(dataset);
617  H5Sclose(filespace);
618  delete[] shape;
619  throw std::runtime_error("Marray cannot get extension of dataset.");
620  }
621 
622  // write shape to out
623  out = std::vector<T>(dimension);
624  if(H5Aexists(dataset, reverseShapeAttributeName) > 0) {
625  for(std::size_t j=0; j<out.size(); ++j) {
626  out[out.size()-j-1] = T(shape[j]);
627  }
628  }
629  else {
630  for(std::size_t j=0; j<out.size(); ++j) {
631  out[j] = T(shape[j]);
632  }
633  }
634 
635  // clean up
636  delete[] shape;
637  H5Dclose(dataset);
638  H5Sclose(filespace);
639  handleCheck.check();
640 }

653 template<class T, class BaseIterator, class ShapeIterator>
654 void loadHyperslab(
655  const hid_t& groupHandle,
656  const std::string& datasetName,
657  BaseIterator baseBegin,
658  BaseIterator baseEnd,
659  ShapeIterator shapeBegin,
660  Marray<T>& out
661 ) {
662  marray_detail::Assert(MARRAY_NO_ARG_TEST || groupHandle >= 0);
663  HandleCheck<MARRAY_NO_DEBUG> handleCheck;
664 
665  // open dataset
666  hid_t dataset = H5Dopen(groupHandle, datasetName.c_str(), H5P_DEFAULT);
667  if(dataset < 0) {
668  throw std::runtime_error("Marray cannot open dataset.");
669  }
670 
671  // determine shape of hyperslab and array
672  std::size_t size = std::distance(baseBegin, baseEnd);
673  std::vector<hsize_t> offset(size);
674  std::vector<hsize_t> slabShape(size);
675  std::vector<hsize_t> marrayShape(size);
676  CoordinateOrder coordinateOrder;
677  if(H5Aexists(dataset, reverseShapeAttributeName) > 0) {
678  // reverse base and shape
679  coordinateOrder = LastMajorOrder;
680  std::size_t j = size-1;
681  std::size_t k = 0;
682  for(;;) {
683  offset[j] = hsize_t(*baseBegin);
684  slabShape[j] = hsize_t(*shapeBegin);
685  marrayShape[k] = slabShape[j];
686  if(j == 0) {
687  break;
688  }
689  else {
690  ++baseBegin;
691  ++shapeBegin;
692  ++k;
693  --j;
694  }
695  }
696  }
697  else {
698  // don't reverse base and shape
699  coordinateOrder = FirstMajorOrder;
700  for(std::size_t j=0; j<size; ++j) {
701  offset[j] = hsize_t(*baseBegin);
702  slabShape[j] = hsize_t(*shapeBegin);
703  marrayShape[j] = slabShape[j];
704  ++baseBegin;
705  ++shapeBegin;
706  }
707  }
708 
709  // select dataspace hyperslab
710  hid_t datatype = H5Dget_type(dataset);
711 
712  if(!H5Tequal(datatype, hdf5Type<T>())) {
713  throw std::runtime_error("data type of stored hdf5 dataset and passed array do not match in loadHyperslab");
714  }
715 
716  hid_t dataspace = H5Dget_space(dataset);
717  herr_t status = H5Sselect_hyperslab(dataspace, H5S_SELECT_SET,
718  &offset[0], NULL, &slabShape[0], NULL);
719  if(status < 0) {
720  H5Tclose(datatype);
721  H5Sclose(dataspace);
722  H5Dclose(dataset);
723  throw std::runtime_error("Marray cannot select hyperslab. Check offset and shape!");
724  }
725 
726  // select memspace hyperslab
727  hid_t memspace = H5Screate_simple(int(size), &marrayShape[0], NULL);
728  std::vector<hsize_t> offsetOut(size, 0); // no offset

768 void
769 saveHyperslab(
770  const hid_t& groupHandle,
771  const std::string& datasetName,
772  BaseIterator baseBegin,
773  BaseIterator baseEnd,
774  ShapeIterator shapeBegin,
775  const Marray<T>& in
776 ) {
777  marray_detail::Assert(MARRAY_NO_ARG_TEST || groupHandle >= 0);
778  HandleCheck<MARRAY_NO_DEBUG> handleCheck;
779 
780  // open dataset
781  hid_t dataset = H5Dopen(groupHandle, datasetName.c_str(), H5P_DEFAULT);
782  if(dataset < 0) {
783  throw std::runtime_error("Marray cannot open dataset.");
784  }
785 
786  // determine hyperslab shape
787  std::vector<hsize_t> memoryShape(in.dimension());
788  for(std::size_t j=0; j<in.dimension(); ++j) {
789  memoryShape[j] = in.shape(j);
790  }
791  std::size_t size = std::distance(baseBegin, baseEnd);
792  std::vector<hsize_t> offset(size);
793  std::vector<hsize_t> slabShape(size);
794  bool reverseShapeAttribute =
795  (H5Aexists(dataset, reverseShapeAttributeName) > 0);
796  if(reverseShapeAttribute && in.coordinateOrder() == LastMajorOrder) {
797  // reverse base and shape
798  std::size_t j = size-1;
799  for(;;) {
800  offset[j] = hsize_t(*baseBegin);
801  slabShape[j] = hsize_t(*shapeBegin);
802  if(j == 0) {
803  break;
804  }
805  else {
806  ++baseBegin;
807  ++shapeBegin;
808  --j;
809  }
810  }
811  }
812  else if(!reverseShapeAttribute && in.coordinateOrder() == FirstMajorOrder) {
813  for(std::size_t j=0; j<size; ++j) {
814  offset[j] = hsize_t(*baseBegin);
815  slabShape[j] = hsize_t(*shapeBegin);
816  ++baseBegin;
817  ++shapeBegin;
818  }
819  }
820  else {
821  H5Dclose(dataset);
822  throw std::runtime_error("Marray cannot write to HDF5 file. A different order was used when the file was created.");
823  }
824 
825  // select dataspace hyperslab
826  hid_t datatype = H5Dget_type(dataset);
827  hid_t dataspace = H5Dget_space(dataset);
828  herr_t status = H5Sselect_hyperslab(dataspace, H5S_SELECT_SET,
829  &offset[0], NULL, &slabShape[0], NULL);
830  if(status < 0) {
831  H5Tclose(datatype);
832  H5Sclose(dataspace);
833  H5Dclose(dataset);
834  throw std::runtime_error("Marray cannot select hyperslab. Check offset and shape!");
835  }
836 
837  // select memspace hyperslab
838  hid_t memspace = H5Screate_simple(int(in.dimension()), &memoryShape[0], NULL);
839  std::vector<hsize_t> memoryOffset(int(in.dimension()), 0); // no offset
840  status = H5Sselect_hyperslab(memspace, H5S_SELECT_SET, &memoryOffset[0], NULL,
841  &memoryShape[0], NULL);
842  if(status < 0) {
843  H5Sclose(memspace);
844  H5Tclose(datatype);
845  H5Sclose(dataspace);
846  H5Dclose(dataset);
847  throw std::runtime_error("Marray cannot select hyperslab. Check offset and shape!");
848  }
849 
850  // write from memspace to dataspace
851  status = H5Dwrite(dataset, datatype, memspace, dataspace, H5P_DEFAULT, &(in(0)));
852 
853  // clean up
854  H5Sclose(memspace);
855  H5Tclose(datatype);
856  H5Sclose(dataspace);
857  H5Dclose(dataset);
858  if(status < 0) {
859  throw std::runtime_error("Marray cannot write to dataset.");
860  }
861  handleCheck.check();
862 }
