148  LiftedMulticutKernighanLin<OBJECTIVE>::
149  optimize(
150  NodeLabelsType & nodeLabels, VisitorBaseType * visitor
151  ){
152 
153  currentBest_ = &nodeLabels;
154 
155  VisitorProxyType visitorProxy(visitor);
156  visitorProxy.begin(this);
157 
158  // build cc on given nodeLabels
159  // and build initial partiton
160  this->initializePartiton();
161 
162 
163  for(const auto node : graph_.nodes()){
164  visited_[node] = 0;
165  }
166 
167  // interatively update bipartition in order to minimize the total cost of the multicut
168  for (size_t k = 0; k < settings_.numberOfOuterIterations; ++k){
169 
170 
171  auto energyDecrease = 0.0;
172 
173  this->buildRegionAdjacencyGraph();
174  this->optimizePairs(energyDecrease);
175  this->introduceNewPartitions(energyDecrease);
176 
177  if(energyDecrease == .0)
178  break;
179 
180  this->connectedComponentLabeling();
181  double newEnergyValue = objective_.evalNodeLabels(twoCutBuffers_.referencedBy);
182 
183 
184  // if the new true energy is higher, than the current one, revert the changes and terminate
185  if (newEnergyValue >= currentBestEnergy_ - settings_.epsilon){
186  for(const auto node : graph_.nodes())
187  nodeLabels[node] = lastGoodVertexLabels_[node];
188  break;
189  }
190 
191  // otherwise, form new partitions
192  this->formNewPartition();
193 
194  for (const auto i : graph_.nodes()){
195  nodeLabels[i] = twoCutBuffers_.vertexLabels[i];
196  }
197 
198  // check if labeling changed
199  if(!hasChanges()){
200  break;
201  }
202 
203  // check if the shape of some partitions didn't change
204  this->checkIfPartitonChanged();
205 
206  // remeber last labeling
207  for(const auto node : graph_.nodes())
208  lastGoodVertexLabels_[node] = twoCutBuffers_.vertexLabels[node];
209 
210  // update best energy
211  currentBestEnergy_ = newEnergyValue;
212 
213  // call visitor
214  if(!visitorProxy.visit(this))
215  break;
216 
217  }
218  visitorProxy.end(this);
219  }
