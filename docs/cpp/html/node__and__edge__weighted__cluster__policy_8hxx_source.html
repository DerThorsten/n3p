187 template<class GRAPH, bool ENABLE_UCM>
188 inline double
189 NodeAndEdgeWeightedClusterPolicy<GRAPH, ENABLE_UCM>::
190 computeWeight(
191  const uint64_t edge
192 ) const {
193  const auto sr = settings_.sizeRegularizer;
194  const auto uv = edgeContractionGraph_.uv(edge);
195  const auto sizeU = nodeSizes_[uv.first];
196  const auto sizeV = nodeSizes_[uv.second];
197  const auto sFac = 2.0 / ( 1.0/std::pow(sizeU,sr) + 1.0/std::pow(sizeV,sr) );
198  const auto fromNodes = this->weightFromNodes(uv.first, uv.second);
199  const auto fromEdge = edgeIndicators_[edge];
200  const auto beta = settings_.beta;
201  const auto e = beta*fromNodes + (1.0-beta)*fromEdge;
202  return e * sFac;
203 }
