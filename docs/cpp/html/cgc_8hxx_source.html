892  template<class OBJECTIVE>
893  void Cgc<OBJECTIVE>::
894  glueAndCutPhase(
895  VisitorProxyType & visitorProxy
896  ){

898 
899  visitorProxy.clearLogNames();
900  visitorProxy.addLogNames({std::string("Sweep"),std::string("be")});
901 
902  // one anchor for all ``cc-edges``
903  typedef std::pair<uint64_t, uint64_t> LabelPair;
904  struct LabelPairHash{
905  public:
906  size_t operator()(const std::pair<uint64_t, uint64_t> & x) const {
907  size_t h = std::hash<uint64_t>()(x.first) ^ std::hash<uint64_t>()(x.second);
908  return h;
909  }
910  };
911  typedef std::unordered_map<LabelPair, uint64_t, LabelPairHash> LabelPairToAnchor;
912  LabelPairToAnchor labelPairToAnchorEdge;
913 
914  // the node labeling as reference
915  auto & nodeLabels = *currentBest_;
916 
917  // initially everything is marked as dirty
918 
919  auto continueSeach = true;
920  auto sweep = 0;
921  while(continueSeach){
922 
923  continueSeach = false;
924 
925 
926  labelPairToAnchorEdge.clear();
927  for(const auto edge : graph_.edges()){
928  const auto u = graph_.u(edge);
929  const auto v = graph_.v(edge);
930  const auto lu = nodeLabels[std::min(u,v)];
931  const auto lv = nodeLabels[std::max(u,v)];
932  if(lu != lv){
933  const auto labelPair = LabelPair(lu, lv);
934  labelPairToAnchorEdge.insert(std::make_pair(labelPair, edge));
935  }
936  }
937 
938  for(const auto kv : labelPairToAnchorEdge){
939  const auto edge = kv.second;
940 
941  if(isDirtyEdge_[edge]){
942 
943  const auto u = graph_.u(edge);
944  const auto v = graph_.v(edge);
945  const auto lu = nodeLabels[u];
946  const auto lv = nodeLabels[v];
947 
948  // if the labels are still different
949  if(lu != lv){
950  const auto ret = submodel_.optimize2(nodeLabels, u, v);
951  if(ret.improvment){
952  continueSeach = true;
953  currentBestEnergy_ -= ret.improvedBy;
954 
955  visitorProxy.setLogValue(0, sweep);
956  visitorProxy.setLogValue(1, currentBestEnergy_);
957  visitorProxy.visit(this);
958  }
959  }
960  }
961  }
962  ++sweep;
963  }
964  }
