894  template<class OBJECTIVE>
895  void Cgc<OBJECTIVE>::
896  glueAndCutPhase(
897  VisitorProxy & visitorProxy
898  ){
899  currentBestEnergy_ = objective_.evalNodeLabels(*currentBest_);
900 
901  visitorProxy.clearLogNames();
902  visitorProxy.addLogNames({std::string("Sweep"),std::string("be")});
903 
904  // one anchor for all ``cc-edges``
905  typedef std::pair<uint64_t, uint64_t> LabelPair;
906  struct LabelPairHash{
907  public:
908  size_t operator()(const std::pair<uint64_t, uint64_t> & x) const {
909  size_t h = std::hash<uint64_t>()(x.first) ^ std::hash<uint64_t>()(x.second);
910  return h;
911  }
912  };
913  typedef std::unordered_map<LabelPair, uint64_t, LabelPairHash> LabelPairToAnchor;
914  LabelPairToAnchor labelPairToAnchorEdge;
915 
916  // the node labeling as reference
917  auto & nodeLabels = *currentBest_;
918 
919  // initially everything is marked as dirty
920 
921  auto continueSeach = true;
922  auto sweep = 0;
923  while(continueSeach){
924 
925  continueSeach = false;
926 
927 
928  labelPairToAnchorEdge.clear();
929  for(const auto edge : graph_.edges()){
930  const auto u = graph_.u(edge);
931  const auto v = graph_.v(edge);
932  const auto lu = nodeLabels[std::min(u,v)];
933  const auto lv = nodeLabels[std::max(u,v)];
934  if(lu != lv){
935  const auto labelPair = LabelPair(lu, lv);
936  labelPairToAnchorEdge.insert(std::make_pair(labelPair, edge));
937  }
938  }
939 
940  for(const auto kv : labelPairToAnchorEdge){
941  const auto edge = kv.second;
942 
943  if(isDirtyEdge_[edge]){
944 
945  const auto u = graph_.u(edge);
946  const auto v = graph_.v(edge);
947  const auto lu = nodeLabels[u];
948  const auto lv = nodeLabels[v];
949 
950  // if the labels are still different
951  if(lu != lv){
952  const auto ret = submodel_.optimize2(nodeLabels, u, v);
953  if(ret.improvment){
954  continueSeach = true;
955  currentBestEnergy_ -= ret.improvedBy;
956 
957  visitorProxy.setLogValue(0, sweep);
958  visitorProxy.setLogValue(1, currentBestEnergy_);
959  visitorProxy.visit(this);
960  }
961  }
962  }
963  }
964  ++sweep;
965  }
966  }
