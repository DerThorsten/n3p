292

295 
296  // search for violated non-chordal cycles and add corresp. inequalities
297  // and for violated cut constraints
298  size_t nCycleConstraints = 0;
299  size_t nCutConstraints = 0;
300 
301  // we iterate over edges and the corresponding lpEdge
302  // for a graph with dense contiguous edge ids the lpEdge
303  // is equivalent to the graph edge
304  auto lpEdge = 0;
305  for (auto edge : liftedGraph_.edges()){
306 
307  const auto uv = liftedGraph_.uv(edge);
308  const auto v0 = uv.first;
309  const auto v1 = uv.second;
310  const auto areConnected = components_.areConnected(v0, v1);
311  const auto ilpLabel = ilpSolver_->label(lpEdge);
312 
313  if (ilpLabel > 0.5 && areConnected){
314 
315  auto hasPath = bibfs_.runSingleSourceSingleTarget(v0, v1, graphSubgraphWithCutTakeUncut);
316  NIFTY_CHECK(hasPath,"internal error");
317  const auto & path = bibfs_.path();
318  NIFTY_CHECK_OP(path.size(),>,0,"");
319  const auto sz = path.size(); //buildPathInLargeEnoughBuffer(v0, v1, bfs.predecessors(), path.begin());
320 
321  bool chordless = true;
322  if (findChord(graph_, graphSubgraphWithCutTakeCut, path.begin(), path.end(), true) != -1){
323  chordless = false;
324  }
325 
326  if(chordless){
327  for (size_t j = 0; j < sz - 1; ++j){
328  const auto v = denseIds_[liftedGraph_.findEdge(path[j], path[j + 1])];
329  NIFTY_ASSERT_OP(v,<,liftedGraph_.numberOfEdges());
330  variables_[j] = v;
331  coefficients_[j] = 1.0;
332  }
333  variables_[sz - 1] = lpEdge;
334  coefficients_[sz - 1] = -1.0;
335 
336  ++addedConstraints_;
337  ilpSolver_->addConstraint(variables_.begin(), variables_.begin() + sz,
338  coefficients_.begin(), 0, std::numeric_limits<double>::infinity());
339  ++nCycleConstraints;
340  }
341 
342  }
343  else if(searchForCutConstraitns && ilpLabel < 0.5 && !areConnected){
344 
345 
346 
347  //std::cout<<"Violated Cut Constraint\n";
348 
349  auto addConstraintFromCut = [&]( const uint64_t va, const uint64_t la){
350 
351  auto nCut = 0;
352  auto dfsVisitor = [&] (
353  int64_t toNode,int64_t predecessorNode,
354  int64_t edge,int64_t distance,
355  bool & continueSeach, bool & addToNode
356  ){
357  if(components_.componentLabel(toNode) != la){
358  coefficients_[nCut] = -1;
359  variables_[nCut] = denseIds_[objective_.graphEdgeInLiftedGraph(edge)];
360  addToNode = false;
361  ++nCut;
362  }
363  };
364  DefaultSubgraphMask<GraphType> subgraphMask;
365  dfs_.run(&va, &va+1, subgraphMask, dfsVisitor);
366 
367  coefficients_[nCut] = 1.0;
368  variables_[nCut] = lpEdge;
369 
370  ilpSolver_->addConstraint(variables_.begin(), variables_.begin() + nCut,
371  coefficients_.begin(), 1.0 - nCut,
372  std::numeric_limits<double>::infinity());
373 
374  //std::cout<<" nCut: "<< nCut<<"\n";
375  };
376 
377  const auto l0 = components_.componentLabel(v0);
378  const auto l1 = components_.componentLabel(v1);
379  addConstraintFromCut(v0, l0);
380  addConstraintFromCut(v1, l1);
381 
382  ++nCutConstraints;
383 
384  }
385 
386  ++lpEdge;
387  }
388 
389  // add additional logs
390  visitorProxy.setLogValue(0, nCycleConstraints);
391  visitorProxy.setLogValue(1, !searchForCutConstraitns ? -1.0 : double(nCutConstraints));
392  // std::cout << "LiftedMulticutIlp::addViolatedInequalities: RETURN\n";
